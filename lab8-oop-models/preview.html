📄 models/Admin.js — Herencia
class Admin extends User {
constructor(name, email, level) {
super(name, email);
this.level = level;
this.permissions = [];
}
 
grantPermission(perm) {
this.permissions.push(perm);
}
 
toString() {
return `Admin: ${this.name}`;
}
}
